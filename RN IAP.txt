iap.zip 파일 RN 프로젝트 루트 밑에 복사

# 루트 경로 기준
# 기존 iap 제거
npm uninstall react-native-iap

# 대상 폴더 비우기
Remove-Item -Recurse -Force .\vendor\react-native-iap-custom -ErrorAction SilentlyContinue
New-Item -ItemType Directory .\vendor | Out-Null

# zip → vendor/react-native-iap-custom 아래로 풀기
tar -xf .\react-native-iap.zip -C .\vendor


# 잘 풀렸는지 확인
Get-ChildItem .\vendor -Depth 2
package.json 이 보여야함

# package.json 열어서 scripts 블럭 삭제

# 설치
npm i .\vendor\react-native-iap --legacy-peer-deps

# 설치 확인
npm ls react-native-iap
